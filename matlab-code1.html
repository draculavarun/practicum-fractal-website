clear
N = 1000000;
r = ceil(rand(N,1)*3);%pick N random numbers between 1 and 3
t = [ 0 0; 1 0;1/2 sqrt(3)/2];%coordinates of three points
p(N,2) = 0;%start at [0 0]

for n=1 :N-1
p(n+1,:)=(t(r(n),:)+p(n,:))/2; % go midway between point chosen at random
end
plot(p(:,1),p(:,2),'.');
axis equal;

%For finding fractal dimension we can add the following code to the above code 
d=exp(-4:0.1:-3)';%precise calculation
C(1:size(d,1),1)=0;
N=size(p,1);%the number of points
Nc=1000;%number of points that were selected at random to average C
r=randperm(N);
for n=1 :Nc
    for m=1 :size(d,1)
        C(m,1)=C(m,1)+sum((p(:,1)-p(r(n),1)).^2+(p(:,2)-p(r(n),2)).^2 < d(m,1)^2);
        %find number of points engulfed
    end
end
C=C/Nc;
figure;
plot(log(d),log(C),'.-');
