clear
p=10;% Sigma Prandtl number
r=28; %Rayleigh number
b=8/3;
x0=[3 3 3];%initial condtion
tspan = [0:0.01:300];% time span
[t,x]=ode45(@(t,x) odefcn(x,p,r,b),tspan,x0);%ode solver
plot3(x(:,1),x(:,2),x(:,3),'LineWidth',1);%plot the solution
axis equal;
set(gca,'FontSize',20);
d=exp(0:0.1:1);
C(1:size(d,1),1)=0;
N=size(x,1);
Nc=1000;
r=randperm(N);
for n=1:Nc
    for m=1:size(d,1)
        C(m,1)=C(m,1)+sum((x(:,1)-x(r(n),1)).^2+(x(:,2)-x(r(n),2)).^2+(x(:,3)-x(r(n),3)).^2 < d(m,1)^2);
    end
end
C=C / Nc;
figure;
plot(log(d),log(C),'.-','MarkerSize',10);
set(gca,'FontSize',20);
%Lorenz Functions
function dxdt=odefcn(x,p,r,b)
dxdt=zeros(2,1);
dxdt(1)=p*(x(2)-x(1));
dxdt(2)=r*x(1)-x(2)-x(1)*x(3);
dxdt(3)=x(1)*x(2)-b*x(3);
end
