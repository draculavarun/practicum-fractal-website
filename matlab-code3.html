clear

p=seed;
plot(p(:,1),p(:,2),'.');
axis equal;
d = exp(2.5:0.02:3)';
C(1:size(d,1),1)=0;
N = size(p,1);
Nc=5000;
r = randperm(N);
for n = 1:Nc
    for m=1:size(d,1)
        C(m,1)=C(m,1)+sum((p(:,1)-p(r(n),1)).^2+(p(:,2)-p(r(n),2)).^2 < d(m,1)^2);
    end
end
C = C/Nc;
figure;
plot(log(d),log(C),'.-');
